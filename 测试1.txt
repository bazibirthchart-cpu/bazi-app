    let currentLang = 'zh';

    function renderLanguage(lang) {
        currentLang = lang;
        
        document.getElementById('title').innerText = i18n[lang].title;
        document.getElementById('subtitle').innerText = i18n[lang].subtitle;
        document.getElementById('lbl-name').innerText = i18n[lang].name;
        document.getElementById('name').placeholder = i18n[lang].namePlh;
        document.getElementById('lbl-gender').innerText = i18n[lang].gender;
        document.getElementById('txt-female').innerText = i18n[lang].female;
        document.getElementById('txt-male').innerText = i18n[lang].male;
        document.getElementById('lbl-birth').innerText = i18n[lang].birth;
        document.getElementById('lbl-dst').innerText = i18n[lang].dst;
        document.getElementById('lbl-location').innerText = i18n[lang].location;
        document.getElementById('btn-submit').innerText = i18n[lang].submit;
        document.getElementById('privacy-text').innerText = i18n[lang].privacy;
        document.getElementById('overseasCity').placeholder = i18n[lang].overseasPlh;

        // 动态修改手写占位符文字
        const birthInput = document.getElementById('birthTime');
        const placeholder = document.getElementById('datePlaceholder');
        if(!birthInput.value) {
            placeholder.innerText = i18n[lang].birthPlh;
            placeholder.style.display = 'block';
        } else {
            placeholder.style.display = 'none';
        }

        const countrySelect = document.getElementById('countrySelect');
        const previousValue = countrySelect.value || 'China';
        countrySelect.innerHTML = `
            <option value="China">${i18n[lang].countryChina}</option>
            <option value="Overseas">${i18n[lang].countryOverseas}</option>
        `;
        countrySelect.value = previousValue;
    }

    // 处理时间选择框的值改变状态
    function handleDateChange(input) {
        const placeholder = document.getElementById('datePlaceholder');
        if (input.value) {
            input.classList.add('has-value');
            placeholder.style.display = 'none';
        } else {
            input.classList.remove('has-value');
            placeholder.innerText = i18n[currentLang].birthPlh;
            placeholder.style.display = 'block';
        }
    }

    function switchLanguage(lang) {
        document.getElementById('btn-lang-zh').classList.toggle('active', lang === 'zh');
        document.getElementById('btn-lang-en').classList.toggle('active', lang === 'en');
        renderLanguage(lang);
        handleCountryChange();
    }

    function handleCountryChange() {
        const country = document.getElementById('countrySelect').value;
        const chinaGroup = document.getElementById('chinaLocationGroup');
        const overseasGroup = document.getElementById('overseasLocationGroup');
        
        if (country === 'China') {
            chinaGroup.style.display = 'grid';
            overseasGroup.style.display = 'none';
            document.getElementById('provinceSelect').required = true;
            document.getElementById('overseasCity').required = false;
        } else {
            chinaGroup.style.display = 'none';
            overseasGroup.style.display = 'block';
            document.getElementById('provinceSelect').required = false;
            document.getElementById('overseasCity').required = true;
        }
    }

    function initChinaLocation() {
        const provinceSelect = document.getElementById('provinceSelect');
        provinceSelect.innerHTML = '';
        if (window.ChinaAreaData && window.ChinaAreaData['86']) {
            const provinces = window.ChinaAreaData['86'];
            for (let code in provinces) {
                let opt = document.createElement('option');
                opt.value = provinces[code];
                opt.dataset.code = code;
                opt.innerText = provinces[code];
                provinceSelect.appendChild(opt);
            }
            loadCities();
        }
    }

    function loadCities() {
        const provinceSelect = document.getElementById('provinceSelect');
        const citySelect = document.getElementById('citySelect');
        if (!provinceSelect.options.length) return;
        
        const selectedOption = provinceSelect.options[provinceSelect.selectedIndex];
        const provinceCode = selectedOption.dataset.code;

        citySelect.innerHTML = '';
        const cities = window.ChinaAreaData[provinceCode];
        for (let code in cities) {
            let opt = document.createElement('option');
            opt.value = cities[code];
            opt.innerText = cities[code];
            citySelect.appendChild(opt);
        }
    }

    window.onload = function() {
        if (window.ChinaAreaData) {
            initChinaLocation();
        }
        const userLang = navigator.language || navigator.userLanguage;
        const defaultLang = userLang.startsWith('zh') ? 'zh' : 'en';
        switchLanguage(defaultLang);
    };

    function handleSubmit(event) {
        event.preventDefault();
        let finalLocation = "";
        const country = document.getElementById('countrySelect').value;
        if (country === 'China') {
            finalLocation = "中国," + document.getElementById('provinceSelect').value + "," + document.getElementById('citySelect').value;
        } else {
            finalLocation = document.getElementById('overseasCity').value;
        }

        const formData = {
            name: document.getElementById('name').value,
            gender: document.querySelector('input[name="gender"]:checked').value,
            birthTime: document.getElementById('birthTime').value,
            isDST: document.getElementById('dst').checked,
            location: finalLocation,
            language: currentLang
        };
        console.log("打包好的数据：", formData);
        alert(currentLang === 'zh' ? "数据准备就绪！即将唤起支付解锁您的专属AI报告。" : "Data ready! Proceeding to payment to unlock your custom AI report.");
    }
